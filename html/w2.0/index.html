;;; This code order a query from an ordering function
(defmethod order-query ((query Query) (order Ordering))
  "Loop over the result set of the query and apply each function of the 
ordering to put result in a new list preceded by its keyword"
  (let ( (res (init-ordering order)) )
    (with-slots (entries) order
      (loop for elt in (result query)
	 do (loop for entry in entries
	       when (funcall (first entry) elt)
	       do (return (push elt (getf res (second entry)))))))
    (setf (result query) res)
    query))

/*
This script detects external links in a page
and attaches a behaviour to them so they open
in a external window.
*/
function initialiseLinks() {
    if (document.getElementsByTagName) {
        var links = document.getElementsByTagName("A");
        for (var i = 0; i < links.length; i++) {
            if (links[i].href.indexOf("http:")==0) {
                // if the links URL starts with http: then we assume it's an external link
                links[i].onclick = function() {
                    window.open(this.href);
                    return false; // stop normal link behaviour
                }
            }
        }
    }
}

window.onload = initialiseLinks();
